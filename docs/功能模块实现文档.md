### （1）项目前端实现相关代码

**文件：`frontend/js/auth.js`**

该文件是全局认证工具对象，负责 Token 与用户信息的本地存储管理，被所有页面引入。

```javascript
// ── 常量：localStorage 存储键名 ──────────────────────────────
const AUTH_KEY = 'mq_token';   // JWT token 的存储键
const USER_KEY = 'mq_user';    // 用户基本信息（nickname/role）的存储键

const Auth = {
  // save(token, user)：登录成功后保存 token 和用户信息到 localStorage
  save(token, user) {
    localStorage.setItem(AUTH_KEY, token);
    localStorage.setItem(USER_KEY, JSON.stringify(user));  // 序列化存储
  },

  // getToken()：读取当前 JWT token，用于 API 请求 Authorization 头
  getToken() { return localStorage.getItem(AUTH_KEY); },

  // getUser()：反序列化读取用户对象，未登录返回 null
  getUser() {
    const u = localStorage.getItem(USER_KEY);
    return u ? JSON.parse(u) : null;
  },

  // isLoggedIn()：判断是否已登录（token 存在即视为登录态）
  isLoggedIn() { return !!this.getToken(); },

  // isAdmin()：判断是否为管理员角色
  isAdmin() { const u = this.getUser(); return u && u.role === 'ADMIN'; },

  // logout()：清除本地存储并跳转到登录页
  logout() {
    localStorage.removeItem(AUTH_KEY);
    localStorage.removeItem(USER_KEY);
    window.location.href = '/login.html';
  },

  // requireLogin()：页面级守卫，未登录则强制跳转登录页
  requireLogin() {
    if (!this.isLoggedIn()) window.location.href = '/login.html';
  },

  // requireAdmin()：管理员页面守卫，非管理员跳转首页
  requireAdmin() {
    if (!this.isAdmin()) window.location.href = '/index.html';
  }
};
```

```javascript
// login()：登录核心函数，读取表单值 → 调用 API → 保存登录态 → 跳转
async function login() {
  const username = document.getElementById('username').value.trim();
  const password = document.getElementById('password').value;

  // 前端基础校验：非空检查
  if (!username || !password) return showToast('请填写用户名和密码', 'error');

  try {
    // 调用后端登录接口，返回 { token, role, nickname }
    const data = await api.login(username, password);

    // 将 token 和用户信息保存到本地存储
    Auth.save(data.token, { nickname: data.nickname, role: data.role });

    showToast('登录成功');
    // 根据角色跳转：管理员进后台，普通用户进首页
    setTimeout(() => window.location.href =
      data.role === 'ADMIN' ? '/admin/index.html' : '/index.html', 800);
  } catch(e) { showToast(e.message, 'error'); }
}
// 支持密码输入框按 Enter 键触发登录：onkeydown="if(event.key==='Enter')login()"
```

```javascript
// register()：注册核心函数，含多项前端校验
async function register() {
  const username = document.getElementById('username').value.trim();
  const password = document.getElementById('password').value;
  const confirmPassword = document.getElementById('confirmPassword').value;
  const nickname = document.getElementById('nickname').value.trim();
  const email = document.getElementById('email').value.trim();

  // 前端校验逻辑（按序执行，遇错即返回）
  if (!username || !password) return showToast('用户名和密码必填', 'error');
  if (username.length < 3)    return showToast('用户名至少3个字符', 'error');
  if (password.length < 6)    return showToast('密码至少6位', 'error');

  try {
    // 调用注册接口，传入全部字段（含 confirmPassword 供后端二次校验）
    await api.register({ username, password, confirmPassword, nickname, email });
    showToast('注册成功，请登录');
    setTimeout(() => window.location.href = '/login.html', 800); // 延迟跳转登录页
  } catch(e) { showToast(e.message, 'error'); }
}
```

**类：`AuthController`**（`controller/AuthController.java`）

REST 控制层，处理认证相关 HTTP 请求，路径前缀 `/api/auth`，全部公开访问（不需 Token）。

```java
@RestController
@RequestMapping("/api/auth")
@RequiredArgsConstructor  // Lombok：自动生成构造函数注入
public class AuthController {
    private final AuthService authService;

    // POST /api/auth/register — 注册接口
    // @Valid 触发 RegisterRequest 中的字段校验注解
    @PostMapping("/register")
    public R<?> register(@Valid @RequestBody RegisterRequest req) {
        authService.register(req);
        return R.ok("注册成功");  // 统一响应包装
    }

    // POST /api/auth/login — 登录接口
    // 返回 Map{ token, role, nickname }
    @PostMapping("/login")
    public R<?> login(@Valid @RequestBody LoginRequest req) {
        return R.ok(authService.login(req));
    }

    // GET /api/auth/me — 获取当前登录用户信息（需携带 JWT）
    // @AuthenticationPrincipal 从 SecurityContext 中注入已认证的 UserDetails
    @GetMapping("/me")
    public R<User> me(@AuthenticationPrincipal UserDetails userDetails) {
        User user = authService.getCurrentUser(userDetails.getUsername());
        user.setPasswordHash(null);  // 脱敏：返回前清除密码哈希
        return R.ok(user);
    }
}
```

**类：`AuthService`**（`service/AuthService.java`）

认证业务逻辑层，负责用户注册、登录验证和当前用户查询。

```java
@Service
@RequiredArgsConstructor
public class AuthService {
    private final UserMapper userMapper;
    private final PasswordEncoder passwordEncoder;  // 密码编码器（MD5）
    private final JwtTokenProvider tokenProvider;   // JWT 工具类

    // register(req)：注册逻辑
    public void register(RegisterRequest req) {
        // 1. 校验用户名唯一性，存在则抛异常（GlobalExceptionHandler 捕获并返回错误响应）
        if (userMapper.selectOne(new QueryWrapper<User>().eq("username", req.getUsername())) != null)
            throw new RuntimeException("用户名已存在");

        // 2. 邮箱非空时校验唯一性
        if (req.getEmail() != null && !req.getEmail().isBlank()
                && userMapper.selectOne(new QueryWrapper<User>().eq("email", req.getEmail())) != null)
            throw new RuntimeException("该邮箱已被注册");

        // 3. 构建用户实体
        User user = new User();
        user.setUsername(req.getUsername());
        user.setPasswordHash(passwordEncoder.encode(req.getPassword())); // 加密密码后存储
        user.setEmail(req.getEmail());
        user.setNickname(req.getNickname() != null ? req.getNickname() : req.getUsername()); // 昵称默认=用户名
        user.setRole("USER");    // 默认角色为普通用户
        user.setIsActive(1);     // 默认账号启用
        userMapper.insert(user);
    }

    // login(req)：登录验证，成功返回 token + 用户基本信息 Map
    public Map<String, String> login(LoginRequest req) {
        User user = userMapper.selectOne(new QueryWrapper<User>().eq("username", req.getUsername()));
        // 用户不存在 或 密码不匹配，统一返回相同错误（防止用户枚举攻击）
        if (user == null || !passwordEncoder.matches(req.getPassword(), user.getPasswordHash()))
            throw new RuntimeException("用户名或密码错误");

        // 生成 JWT，将 username 和 role 写入 Payload
        String token = tokenProvider.generateToken(user.getUsername(), user.getRole());
        return Map.of("token", token, "role", user.getRole(),
                "nickname", user.getNickname() != null ? user.getNickname() : user.getUsername());
    }

    // getCurrentUser(username)：按用户名查询用户实体（供 /me 接口使用）
    public User getCurrentUser(String username) {
        return userMapper.selectOne(new QueryWrapper<User>().eq("username", username));
    }
}
```

**类：`JwtTokenProvider`**（`security/JwtTokenProvider.java`）

JWT 工具类，负责 Token 的生成、验证和信息提取。

```java
@Component
public class JwtTokenProvider {
    private final SecretKey key;     // HMAC-SHA 密钥（从配置加载）
    private final long expiration;   // Token 有效期（毫秒，从配置加载）

    // 构造时从 application.yml 读取 jwt.secret 和 jwt.expiration 配置
    public JwtTokenProvider(
            @Value("${jwt.secret}") String secret,
            @Value("${jwt.expiration}") long expiration) {
        // 用配置的 secret 字符串构造 HMAC-SHA 密钥
        this.key = Keys.hmacShaKeyFor(secret.getBytes(StandardCharsets.UTF_8));
        this.expiration = expiration;
    }

    // generateToken：生成 JWT，Payload 含 username(sub)、role 自定义 Claim、签发/过期时间
    public String generateToken(String username, String role) {
        return Jwts.builder()
                .subject(username)
                .claim("role", role)          // 存入角色信息
                .issuedAt(new Date())         // 签发时间
                .expiration(new Date(System.currentTimeMillis() + expiration)) // 过期时间
                .signWith(key)
                .compact();
    }

    // validateToken：验证签名与过期时间，任何异常均返回 false（不向上抛异常）
    public boolean validateToken(String token) {
        try {
            Jwts.parser().verifyWith(key).build().parseSignedClaims(token);
            return true;
        } catch (JwtException | IllegalArgumentException e) { return false; }
    }

    // getUsername：从已验证的 Token Payload 中提取用户名（subject 字段）
    public String getUsername(String token) {
        return Jwts.parser().verifyWith(key).build()
                .parseSignedClaims(token).getPayload().getSubject();
    }

    // getRole：从已验证的 Token Payload 中提取角色（自定义 claim）
    public String getRole(String token) {
        return (String) Jwts.parser().verifyWith(key).build()
                .parseSignedClaims(token).getPayload().get("role");
    }
}
```

**类：`JwtAuthFilter`**（`security/JwtAuthFilter.java`）

JWT 认证过滤器，继承 `OncePerRequestFilter`，每个请求执行一次，将合法 Token 解析为 Spring Security 认证对象。

```java
@Component
@RequiredArgsConstructor
public class JwtAuthFilter extends OncePerRequestFilter {

    @Override
    protected void doFilterInternal(HttpServletRequest req, HttpServletResponse res,
                                    FilterChain chain) throws ServletException, IOException {
        String header = req.getHeader("Authorization");  // 读取请求头

        // 1. 提取 Bearer Token（格式：Authorization: Bearer <token>）
        if (header != null && header.startsWith("Bearer ")) {
            String token = header.substring(7);  // 截取 "Bearer " 后的 token 字符串

            if (tokenProvider.validateToken(token)) {
                // 2. Token 合法：加载用户详情并构建认证对象
                String username = tokenProvider.getUsername(token);
                UserDetails ud = userDetailsService.loadUserByUsername(username);

                // 3. 将认证信息写入 SecurityContext（后续接口可通过 @AuthenticationPrincipal 注入）
                var auth = new UsernamePasswordAuthenticationToken(ud, null, ud.getAuthorities());
                SecurityContextHolder.getContext().setAuthentication(auth);
            }
        }
        chain.doFilter(req, res);  // 继续过滤器链
    }
}
```

Spring Security 访问控制配置，定义各路径的权限规则。

```java
@Configuration
@EnableWebSecurity
public class SecurityConfig {

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            .cors(Customizer.withDefaults())    // 启用 CORS，由 WebMvcConfig 配置跨域规则
            .csrf(csrf -> csrf.disable())       // 前后端分离模式禁用 CSRF
            .sessionManagement(sm -> sm.sessionCreationPolicy(SessionCreationPolicy.STATELESS)) // 无状态会话
            .authorizeHttpRequests(auth -> auth
                .requestMatchers("/api/auth/**").permitAll()                        // 注册/登录接口公开
                .requestMatchers(HttpMethod.GET, "/api/cats/**", "/api/news/**").permitAll() // 猫咪/新闻GET公开
                .requestMatchers("/api/admin/**").hasRole("ADMIN")                  // 后台接口仅管理员可访问
                .requestMatchers("/uploads/**").permitAll()                         // 上传文件目录公开
                .anyRequest().authenticated()                                       // 其他接口需登录
            )
            .addFilterBefore(jwtAuthFilter, UsernamePasswordAuthenticationFilter.class); // JWT 过滤器前置
        return http.build();
    }

    // passwordEncoder：自定义 MD5 密码编码器
    @Bean
    public PasswordEncoder passwordEncoder() {
        return new PasswordEncoder() {
            @Override
            public String encode(CharSequence raw) {
                // 使用 MD5 算法对密码进行哈希，返回十六进制字符串
                MessageDigest md = MessageDigest.getInstance("MD5");
                byte[] bytes = md.digest(raw.toString().getBytes(StandardCharsets.UTF_8));
                StringBuilder sb = new StringBuilder();
                for (byte b : bytes) sb.append(String.format("%02x", b));
                return sb.toString();
            }
            @Override
            public boolean matches(CharSequence raw, String encoded) {
                return encode(raw).equals(encoded);  // 对比编码结果
            }
        };
    }
}
```

```javascript
async function loadCats(page) {
  const params = {
    page,
    size: 12,                                               // 每页12条
```

```java
@RestController
@RequestMapping("/api/cats")
@RequiredArgsConstructor
public class CatController {
    private final CatService catService;
    private final UserMapper userMapper;

    // GET /api/cats — 分页查询已审核通过的猫咪（公开）
    // 支持 name 模糊搜索、breed 模糊搜索、sort 排序（latest/likes）
    @GetMapping
    public R<?> list(@RequestParam(defaultValue = "1") int page,
                     @RequestParam(defaultValue = "12") int size,
                     @RequestParam(required = false) String name,
                     @RequestParam(required = false) String breed,
                     @RequestParam(defaultValue = "latest") String sort) {
        return R.ok(catService.listApproved(page, size, name, breed, sort));
    }

    // GET /api/cats/{id} — 查询单只猫咪详情（公开），不存在或未审核则抛异常
    @GetMapping("/{id}")
    public R<Cat> detail(@PathVariable Long id) {
        return R.ok(catService.getById(id));
    }

    // POST /api/cats — 提交猫咪（需登录）
    // 从 SecurityContext 获取当前用户，作为提交人 submitterId
    @PostMapping
    public R<?> submit(@Valid @RequestBody CatSubmitRequest req,
                       @AuthenticationPrincipal UserDetails userDetails) {
        User user = userMapper.selectOne(new QueryWrapper<User>().eq("username", userDetails.getUsername()));
        catService.submitCat(req, user.getId());
        return R.ok("提交成功，等待审核");
    }

    // POST /api/cats/{id}/like — 点赞/取消点赞（需登录，幂等操作）
    // 返回 liked（是否已点赞）和 likeCount（当前点赞数）
    @PostMapping("/{id}/like")
    public R<?> like(@PathVariable Long id,
                     @AuthenticationPrincipal UserDetails userDetails) {
        User user = userMapper.selectOne(new QueryWrapper<User>().eq("username", userDetails.getUsername()));
        return R.ok(catService.toggleLike(id, user.getId()));
    }
}
```

```java
@Service
@RequiredArgsConstructor
public class CatService {
    private final CatMapper catMapper;
    private final CatLikeMapper catLikeMapper;  // 点赞关联表 Mapper

    // listApproved：查询 status=APPROVED 的猫咪，支持多条件过滤与排序
    public Page<Cat> listApproved(int page, int size, String name, String breed, String sort) {
        QueryWrapper<Cat> qw = new QueryWrapper<Cat>().eq("status", "APPROVED"); // 只查已审核
        if (StringUtils.hasText(name))  qw.like("name", name);    // 姓名模糊查询
        if (StringUtils.hasText(breed)) qw.like("breed", breed);  // 品种模糊查询
        if ("likes".equals(sort)) qw.orderByDesc("like_count");   // 按点赞数降序
        else qw.orderByDesc("created_at");                         // 默认按创建时间降序
        return catMapper.selectPage(new Page<>(page, size), qw);  // MybatisPlus 分页查询
    }

    // getById：根据 ID 查询，未找到或未审核通过时抛异常
    public Cat getById(Long id) {
        Cat cat = catMapper.selectById(id);
        if (cat == null || !"APPROVED".equals(cat.getStatus())) throw new RuntimeException("猫咪不存在");
        return cat;
    }

    // submitCat：构建 Cat 实体并入库，状态默认 PENDING 待审核
    public void submitCat(CatSubmitRequest req, Long submitterId) {
        Cat cat = new Cat();
        cat.setName(req.getName());
        cat.setGender(req.getGender());
        cat.setAge(req.getAge());
        cat.setBreed(req.getBreed());
        cat.setAvatarUrl(req.getAvatarUrl());
        cat.setPersonality(req.getPersonality());
        cat.setLikeCount(0);          // 初始点赞数为 0
        cat.setStatus("PENDING");     // 提交后默认为待审核状态
        cat.setSubmitterId(submitterId);
        catMapper.insert(cat);
    }

    // toggleLike：点赞/取消点赞的幂等操作
    // 使用 @Transactional 保证 cat_like 表和 cat 表 like_count 的原子一致性
    @Transactional
    public Map<String, Object> toggleLike(Long catId, Long userId) {
        QueryWrapper<CatLike> qw = new QueryWrapper<CatLike>()
                .eq("cat_id", catId).eq("user_id", userId);
        CatLike existing = catLikeMapper.selectOne(qw);
        boolean liked;
        if (existing == null) {
            // 未点赞 → 插入点赞记录 + like_count 加 1（setSql 实现数据库原子自增）
            CatLike like = new CatLike();
            like.setCatId(catId); like.setUserId(userId);
            catLikeMapper.insert(like);
            catMapper.update(null, new UpdateWrapper<Cat>()
                    .eq("id", catId).setSql("like_count = like_count + 1"));
            liked = true;
        } else {
            // 已点赞 → 删除点赞记录 + like_count 减 1
            catLikeMapper.delete(qw);
            catMapper.update(null, new UpdateWrapper<Cat>()
                    .eq("id", catId).setSql("like_count = like_count - 1"));
            liked = false;
        }
        Cat cat = catMapper.selectById(catId);
        return Map.of("liked", liked, "likeCount", cat.getLikeCount());  // 返回最新状态
    }
}
```

```java
    public void blockComment(Long commentId) {
        Comment c = new Comment();
        c.setId(commentId);
        c.setIsBlocked(1);  // 标记为已屏蔽
        commentMapper.updateById(c);
    }
}
```

**类：`FileService`**（`service/FileService.java`）

文件存储业务逻辑，负责生成唯一文件名、创建目录、写入文件。

```java
@Service
public class FileService {
    @Value("${upload.dir}")    // 从 application.yml 读取上传目录（如 ./uploads）
    private String uploadDir;

    // save：生成 UUID 文件名 → 创建目录 → 写入文件 → 返回访问路径
    public String save(MultipartFile file) throws IOException {
        String ext = getExtension(file.getOriginalFilename());  // 保留原始文件扩展名
        String filename = UUID.randomUUID() + ext;              // UUID 命名，防止文件名冲突
        Path dir = Paths.get(uploadDir);
        Files.createDirectories(dir);                           // 目录不存在时自动创建
        Files.copy(file.getInputStream(), dir.resolve(filename)); // 将文件流写入磁盘
        return "/uploads/" + filename;                           // 返回前端可直接访问的相对 URL
    }

    // getExtension：提取文件扩展名，文件名为 null 或无扩展名时默认 .jpg
    private String getExtension(String filename) {
        if (filename == null) return ".jpg";
        int dot = filename.lastIndexOf('.');
        return dot >= 0 ? filename.substring(dot) : ".jpg";
    }
}
```
